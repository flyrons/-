tasks = []

def add_task():
    """–î–æ–±–∞–≤–ª—è–µ—Ç –Ω–æ–≤—É—é –∑–∞–¥–∞—á—É"""
    task_name = input("–í–≤–µ–¥–∏—Ç–µ –Ω–∞–∑–≤–∞–Ω–∏–µ –∑–∞–¥–∞—á–∏: ")
    tasks.append({"name": task_name, "completed": False})

    else:
        print("–°–ø–∏—Å–æ–∫ –∑–∞–¥–∞—á:")
        for i, task in enumerate(tasks):
            status = "‚úîÔ∏è –í—ã–ø–æ–ª–Ω–µ–Ω–∞" if task["completed"] else "üî¥ –ù–µ –≤—ã–ø–æ–ª–Ω–µ–Ω–∞"

            print("–ó–∞–¥–∞—á–∞ –æ—Ç–º–µ—á–µ–Ω–∞ –∫–∞–∫ –≤—ã–ø–æ–ª–Ω–µ–Ω–Ω–∞—è.")
        else:
